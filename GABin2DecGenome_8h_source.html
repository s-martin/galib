6  DESCRIPTION:
7  This header defines the interface for the binary-to-decimal genome and the
8 phenotype objects used for the binary-to-decimal mappings in the
9 binary-to-decimal genomes.
10  The binary-to-decimal phenotype uses reference counting so that you only
11 need one instance of the underlying bit-to-float mapping. Whenever you create
12 a new phenotype it gets its own core. You can use the 'link' member to make
13 a phenotype refer to another's core. This is what the genomes do when they
14 clone themselves so that you don't end up with multiple phenotypes running
15 around. The core will persist until the last phenotype is destroyed (last one
16 out, please turn off the lights).
17  When you clone a phenotype, you get a competeley new one with its own core.

58 class GABin2DecPhenotype
59 {
60  public:
61  GABin2DecPhenotype() : core(new GABin2DecPhenotypeCore) {}
62  GABin2DecPhenotype(const GABin2DecPhenotype &p)
63  : core(new GABin2DecPhenotypeCore(*(p.core)))
64  {
65  }
66  virtual ~GABin2DecPhenotype()
67  {
68  core->cnt -= 1;
69  if (core->cnt == 0)
70  {
71  delete core;
72  }
73  }
74  GABin2DecPhenotype &operator=(const GABin2DecPhenotype &p)
75  {
76  if (&p != this)
77  {
78  *core = *(p.core);
79  }
80  return *this;
81  }
82  GABin2DecPhenotype *clone() const { return new GABin2DecPhenotype(*this); }
83  void link(GABin2DecPhenotype &p)
84  {
85  core->cnt -= 1;
86  if (core->cnt == 0)
87  {
88  delete core;
89  }
90  core = p.core;
91  core->cnt += 1;
92  }
93  void add(unsigned int nbits, float min, float max);
94  void remove(unsigned int which);
95  unsigned int size() const { return core->sz; }
96  unsigned int nPhenotypes() const { return core->n; }
97  float min(int which) const { return core->minval[which]; }
98  float max(int which) const { return core->maxval[which]; }
99  int length(int which) const { return core->nbits[which]; }
100  int offset(int which) const { return core->oset[which]; }
101  bool equal(const GABin2DecPhenotype &) const;
102 
103  protected:
104  GABin2DecPhenotypeCore *core;
105 };
