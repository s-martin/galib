32  static int SwapMutator(GAGenome &c, float pmut)
33  {
34  GA2DArrayGenome<T> &child = DYN_CAST(GA2DArrayGenome<T> &, c);
35 
36  if (pmut <= 0.0)
37  return (0);
38 
39  float nMut = pmut * STA_CAST(float, child.size());
40  int size = child.size() - 1;
41  if (nMut < 1.0)
42  { // we have to do a flip test on each bit
43  nMut = 0;
44  for (int i = size; i >= 0; i--)
45  {
46  if (GAFlipCoin(pmut))
47  {
48  child.GAArray<T>::swap(i, GARandomInt(0, size));
49  nMut++;
50  }
51  }
52  }
53  else
54  { // only flip the number of bits we need to flip
55  for (int n = 0; n < nMut; n++)
56  child.GAArray<T>::swap(GARandomInt(0, size),
57  GARandomInt(0, size));
58  }
59  return (STA_CAST(int, nMut));
60  }

62  static float ElementComparator(const GAGenome &a, const GAGenome &b)
63  {
64  const GA2DArrayGenome<T> &sis = DYN_CAST(const GA2DArrayGenome<T> &, a);
65  const GA2DArrayGenome<T> &bro = DYN_CAST(const GA2DArrayGenome<T> &, b);
66  if (sis.size() != bro.size())
67  return -1;
68  if (sis.size() == 0)
69  return 0;
70  float count = 0.0;
71  for (int i = sis.width() - 1; i >= 0; i--)
72  for (int j = sis.height() - 1; j >= 0; j--)
73  count += ((sis.gene(i, j) == bro.gene(i, j)) ? 0 : 1);
74  return count / sis.size();
75  }

77  static int UniformCrossover(const GAGenome &p1, const GAGenome &p2,
78  GAGenome *c1, GAGenome *c2)
79  {
80  const GA2DArrayGenome<T> &mom =
81  DYN_CAST(const GA2DArrayGenome<T> &, p1);
82  const GA2DArrayGenome<T> &dad =
83  DYN_CAST(const GA2DArrayGenome<T> &, p2);
84 
85  int nc = 0;
86  int i, j;
87 
88  if (c1 && c2)
89  {
90  GA2DArrayGenome<T> &sis = DYN_CAST(GA2DArrayGenome<T> &, *c1);
91  GA2DArrayGenome<T> &bro = DYN_CAST(GA2DArrayGenome<T> &, *c2);
92 
93  if (sis.width() == bro.width() && sis.height() == bro.height() &&
94  mom.width() == dad.width() && mom.height() == dad.height() &&
95  sis.width() == mom.width() && sis.height() == mom.height())
96  {
97  for (i = sis.width() - 1; i >= 0; i--)
98  {
99  for (j = sis.height() - 1; j >= 0; j--)
100  {
101  if (GARandomBit())
102  {
103  sis.gene(i, j, mom.gene(i, j));
104  bro.gene(i, j, dad.gene(i, j));
105  }
106  else
107  {
108  sis.gene(i, j, dad.gene(i, j));
109  bro.gene(i, j, mom.gene(i, j));
110  }
111  }
112  }
113  }
114  else
115  {
116  GAMask mask;
117  int startx, starty;
118  int maxx =
119  (sis.width() > bro.width()) ? sis.width() : bro.width();
120  int minx =
121  (mom.width() < dad.width()) ? mom.width() : dad.width();
122  int maxy =
123  (sis.height() > bro.height()) ? sis.height() : bro.height();
124  int miny =
125  (mom.height() < dad.height()) ? mom.height() : dad.height();
126  mask.size(maxx * maxy);
127  for (i = 0; i < maxx; i++)
128  for (j = 0; j < maxy; j++)
129  mask[i * maxy + j] = GARandomBit();
130  startx = (sis.width() < minx) ? sis.width() : minx;
131  starty = (sis.height() < miny) ? sis.height() : miny;
132  for (i = startx - 1; i >= 0; i--)
133  for (j = starty - 1; j >= 0; j--)
134  sis.gene(i, j,
135  (mask[i * starty + j] ? mom.gene(i, j)
136  : dad.gene(i, j)));
137  startx = (bro.width() < minx) ? bro.width() : minx;
138  starty = (bro.height() < miny) ? bro.height() : miny;
139  for (i = startx - 1; i >= 0; i--)
140  for (j = starty - 1; j >= 0; j--)
141  bro.gene(i, j,
142  (mask[i * starty + j] ? dad.gene(i, j)
143  : mom.gene(i, j)));
144  }
145 
146  nc = 2;
147  }
148  else if (c1)
149  {
150  GA2DArrayGenome<T> &sis = DYN_CAST(GA2DArrayGenome<T> &, *c1);
151 
152  if (mom.width() == dad.width() && mom.height() == dad.height() &&
153  sis.width() == mom.width() && sis.height() == mom.height())
154  {
155  for (i = sis.width() - 1; i >= 0; i--)
156  for (j = sis.height() - 1; j >= 0; j--)
157  sis.gene(
158  i, j,
159  (GARandomBit() ? mom.gene(i, j) : dad.gene(i, j)));
160  }
161  else
162  {
163  int minx =
164  (mom.width() < dad.width()) ? mom.width() : dad.width();
165  int miny =
166  (mom.height() < dad.height()) ? mom.height() : dad.height();
167  minx = (sis.width() < minx) ? sis.width() : minx;
168  miny = (sis.height() < miny) ? sis.height() : miny;
169  for (i = minx - 1; i >= 0; i--)
170  for (j = miny - 1; j >= 0; j--)
171  sis.gene(
172  i, j,
173  (GARandomBit() ? mom.gene(i, j) : dad.gene(i, j)));
174  }
175 
176  nc = 1;
177  }
178 
179  return nc;
180  }

182  // This crossover does clipping (no padding) for resizables. Notice that
183  // this means that any resizable children of two parents will have identical

185  static int OnePointCrossover(const GAGenome &p1, const GAGenome &p2,
186  GAGenome *c1, GAGenome *c2)
187  {
188  const GA2DArrayGenome<T> &mom =
189  DYN_CAST(const GA2DArrayGenome<T> &, p1);
190  const GA2DArrayGenome<T> &dad =
191  DYN_CAST(const GA2DArrayGenome<T> &, p2);
192 
193  int nc = 0;
194  unsigned int momsitex, momlenx, momsitey, momleny;
195  unsigned int dadsitex, dadlenx, dadsitey, dadleny;
196  unsigned int sitex, lenx, sitey, leny;
197 
198  if (c1 && c2)
199  {
200  GA2DArrayGenome<T> &sis = DYN_CAST(GA2DArrayGenome<T> &, *c1);
201  GA2DArrayGenome<T> &bro = DYN_CAST(GA2DArrayGenome<T> &, *c2);
202 
203  if (sis.resizeBehaviour(Dimension::WIDTH) == GAGenome::FIXED_SIZE &&
204  bro.resizeBehaviour(Dimension::WIDTH) == GAGenome::FIXED_SIZE)
205  {
206  if (mom.width() != dad.width() || sis.width() != bro.width() ||
207  sis.width() != mom.width())
208  {
209  GAErr(GA_LOC, mom.className(), "one-point cross",
210  GAError::SameLengthReqd);
211  return nc;
212  }
213  sitex = momsitex = dadsitex = GARandomInt(0, mom.width());
214  lenx = momlenx = dadlenx = mom.width() - momsitex;
215  }
216  else if (sis.resizeBehaviour(Dimension::WIDTH) ==
217  GAGenome::FIXED_SIZE ||
218  bro.resizeBehaviour(Dimension::WIDTH) ==
219  GAGenome::FIXED_SIZE)
220  {
221  GAErr(GA_LOC, mom.className(), "one-point cross",
222  GAError::SameBehavReqd);
223  return nc;
224  }
225  else
226  {
227  momsitex = GARandomInt(0, mom.width());
228  dadsitex = GARandomInt(0, dad.width());
229  momlenx = mom.width() - momsitex;
230  dadlenx = dad.width() - dadsitex;
231  sitex = GAMin(momsitex, dadsitex);
232  lenx = GAMin(momlenx, dadlenx);
233  }
234 
235  if (sis.resizeBehaviour(Dimension::HEIGHT) ==
236  GAGenome::FIXED_SIZE &&
237  bro.resizeBehaviour(Dimension::HEIGHT) == GAGenome::FIXED_SIZE)
238  {
239  if (mom.height() != dad.height() ||
240  sis.height() != bro.height() ||
241  sis.height() != mom.height())
242  {
243  GAErr(GA_LOC, mom.className(), "one-point cross",
244  GAError::SameLengthReqd);
245  return nc;
246  }
247  sitey = momsitey = dadsitey = GARandomInt(0, mom.height());
248  leny = momleny = dadleny = mom.height() - momsitey;
249  }
250  else if (sis.resizeBehaviour(Dimension::HEIGHT) ==
251  GAGenome::FIXED_SIZE ||
252  bro.resizeBehaviour(Dimension::HEIGHT) ==
253  GAGenome::FIXED_SIZE)
254  {
255  GAErr(GA_LOC, mom.className(), "one-point cross",
256  GAError::SameBehavReqd);
257  return nc;
258  }
259  else
260  {
261  momsitey = GARandomInt(0, mom.height());
262  dadsitey = GARandomInt(0, dad.height());
263  momleny = mom.height() - momsitey;
264  dadleny = dad.height() - dadsitey;
265  sitey = GAMin(momsitey, dadsitey);
266  leny = GAMin(momleny, dadleny);
267  }
268 
269  sis.resize(sitex + lenx, sitey + leny);
270  bro.resize(sitex + lenx, sitey + leny);
271 
272  sis.copy(mom, 0, 0, momsitex - sitex, momsitey - sitey, sitex,
273  sitey);
274  sis.copy(dad, sitex, 0, dadsitex, dadsitey - sitey, lenx, sitey);
275  sis.copy(dad, 0, sitey, dadsitex - sitex, dadsitey, sitex, leny);
276  sis.copy(mom, sitex, sitey, momsitex, momsitey, lenx, leny);
277 
278  bro.copy(dad, 0, 0, dadsitex - sitex, dadsitey - sitey, sitex,
279  sitey);
280  bro.copy(mom, sitex, 0, momsitex, momsitey - sitey, lenx, sitey);
281  bro.copy(mom, 0, sitey, momsitex - sitex, momsitey, sitex, leny);
282  bro.copy(dad, sitex, sitey, dadsitex, dadsitey, lenx, leny);
283 
284  nc = 2;
285  }
286  else if (c1)
287  {
288  GA2DArrayGenome<T> &sis = DYN_CAST(GA2DArrayGenome<T> &, *c1);
289 
290  if (sis.resizeBehaviour(Dimension::WIDTH) == GAGenome::FIXED_SIZE)
291  {
292  if (mom.width() != dad.width() || sis.width() != mom.width())
293  {
294  GAErr(GA_LOC, mom.className(), "one-point cross",
295  GAError::SameLengthReqd);
296  return nc;
297  }
298  sitex = momsitex = dadsitex = GARandomInt(0, mom.width());
299  lenx = momlenx = dadlenx = mom.width() - momsitex;
300  }
301  else
302  {
303  momsitex = GARandomInt(0, mom.width());
304  dadsitex = GARandomInt(0, dad.width());
305  momlenx = mom.width() - momsitex;
306  dadlenx = dad.width() - dadsitex;
307  sitex = GAMin(momsitex, dadsitex);
308  lenx = GAMin(momlenx, dadlenx);
309  }
310 
311  if (sis.resizeBehaviour(Dimension::HEIGHT) == GAGenome::FIXED_SIZE)
312  {
313  if (mom.height() != dad.height() ||
314  sis.height() != mom.height())
315  {
316  GAErr(GA_LOC, mom.className(), "one-point cross",
317  GAError::SameLengthReqd);
318  return nc;
319  }
320  sitey = momsitey = dadsitey = GARandomInt(0, mom.height());
321  leny = momleny = dadleny = mom.height() - momsitey;
322  }
323  else
324  {
325  momsitey = GARandomInt(0, mom.height());
326  dadsitey = GARandomInt(0, dad.height());
327  momleny = mom.height() - momsitey;
328  dadleny = dad.height() - dadsitey;
329  sitey = GAMin(momsitey, dadsitey);
330  leny = GAMin(momleny, dadleny);
331  }
332 
333  sis.resize(sitex + lenx, sitey + leny);
334 
335  if (GARandomBit())
336  {
337  sis.copy(mom, 0, 0, momsitex - sitex, momsitey - sitey, sitex,
338  sitey);
339  sis.copy(dad, sitex, 0, dadsitex, dadsitey - sitey, lenx,
340  sitey);
341  sis.copy(dad, 0, sitey, dadsitex - sitex, dadsitey, sitex,
342  leny);
343  sis.copy(mom, sitex, sitey, momsitex, momsitey, lenx, leny);
344  }
345  else
346  {
347  sis.copy(dad, 0, 0, dadsitex - sitex, dadsitey - sitey, sitex,
348  sitey);
349  sis.copy(mom, sitex, 0, momsitex, momsitey - sitey, lenx,
350  sitey);
351  sis.copy(mom, 0, sitey, momsitex - sitex, momsitey, sitex,
352  leny);
353  sis.copy(dad, sitex, sitey, dadsitex, dadsitey, lenx, leny);
354  }
355 
356  nc = 1;
357  }
358 
359  return nc;
360  }

516  void copy(const GA2DArrayGenome<T> &orig, unsigned int r, unsigned int s,
517  unsigned int x, unsigned int y, unsigned int w, unsigned int h)
518  {
519  if (w == 0 || x >= orig.nx || r >= nx || h == 0 || y >= orig.ny ||
520  s >= ny)
521  return;
522  if (x + w > orig.nx)
523  w = orig.nx - x;
524  if (y + h > orig.ny)
525  h = orig.ny - y;
526  if (r + w > nx)
527  w = nx - r;
528  if (s + h > ny)
529  h = ny - s;
530 
531  for (unsigned int j = 0; j < h; j++)
532  GAArray<T>::copy(orig, (s + j) * nx + r, (y + j) * orig.nx + x, w);
533 
534  _evaluated = false;

555  virtual int resize(int w, int h)
556  {
557  if (w == STA_CAST(int, nx) && h == STA_CAST(int, ny))
558  return this->size();
559 
560  if (w == GAGenome::ANY_SIZE)
561  w = GARandomInt(minX, maxX);
562  else if (w < 0)
563  w = nx; // do nothing
564  else if (minX == maxX)
565  minX = maxX = w;
566  else
567  {
568  if (w < STA_CAST(int, minX))
569  w = minX;
570  if (w > STA_CAST(int, maxX))
571  w = maxX;
572  }
573 
574  if (h == GAGenome::ANY_SIZE)
575  h = GARandomInt(minY, maxY);
576  else if (h < 0)
577  h = ny; // do nothing
578  else if (minY == maxY)
579  minY = maxY = h;
580  else
581  {
582  if (h < STA_CAST(int, minY))
583  h = minY;
584  if (h > STA_CAST(int, maxY))
585  h = maxY;
586  }
587 
588  if (w < STA_CAST(int, nx))
589  {
590  int y = GAMin(STA_CAST(int, ny), h);
591  for (int j = 0; j < y; j++)
592  GAArray<T>::move(j * w, j * nx, w);
593  }
594  GAArray<T>::size(w * h);
595  if (w > STA_CAST(int, nx))
596  { // adjust the existing chunks of bits
597  int y = GAMin(STA_CAST(int, ny), h);
598  for (int j = y - 1; j >= 0; j--)
599  GAArray<T>::move(j * w, j * nx, nx);
600  }
601 
602  nx = w;
603  ny = h;
604  _evaluated = false;

761  {
762  GA2DArrayAlleleGenome<T> &child =
763  DYN_CAST(GA2DArrayAlleleGenome<T> &, c);
764

789  int i = m % child.width();
790  int j = m / child.width();
